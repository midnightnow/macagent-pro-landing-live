function rankProblems(kb, text) {
    const t = ' ' + text.toLowerCase() + ' ';
    return kb.map(p => {
        const hits = p.keywords.reduce((n, k) => n + (t.includes(k.toLowerCase()) ? 1 : 0), 0);
        return { p, score: hits };
    }).sort((a, b) => b.score - a.score);
}
